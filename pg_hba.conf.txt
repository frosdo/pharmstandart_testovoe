# PostgreSQL Client Authentication Configuration File
# ===================================================
# Refer to the "Client Authentication" section in the
# PostgreSQL documentation for a complete description
# of this file.  A short synopsis follows.
#
# This file controls: which hosts are allowed to connect,
# how clients are authenticated, which PostgreSQL user
# names they can use, which databases they can access.
#
# Records take one of these forms:
#
# local      DATABASE  USER  METHOD  [OPTIONS]
# host       DATABASE  USER  ADDRESS  METHOD  [OPTIONS]
# hostssl    DATABASE  USER  ADDRESS  METHOD  [OPTIONS]
# hostnossl  DATABASE  USER  ADDRESS  METHOD  [OPTIONS]
#
# (The uppercase items must be replaced by actual values.)
#
# The first field is the connection type:
# - "local" is a Unix-domain socket
# - "host" is a TCP/IP socket (encrypted or not)
# - "hostssl" is a TCP/IP socket that is SSL-encrypted
# - "hostnossl" is a plain TCP/IP socket
#
# DATABASE can be "all", "sameuser", "samerole", "replication",
# a database name, or a comma-separated list thereof.
#
# USER can be "all", a user name, a group name prefixed with "+",
# or a comma-separated list thereof.  In both DATABASE and USER
# fields you can also write a file name prefixed with "@" to include
# names from a separate file.
#
# ADDRESS specifies the client machine address(es) that can match
# this record.  It can contain an IP address, an IP prefix (CIDR),
# or a hostname (only if not using hostname-based authentication).
#
# METHOD can be "trust", "reject", "md5", "password", "scram-sha-256",
# "gss", "sspi", "ident", "peer", "pam", "ldap", "radius", "cert",
# or "bsd".  See the PostgreSQL documentation for details.
#
# This file is read on server startup and when the server receives a
# SIGHUP signal.  If you edit the file on a running system, you have
# to SIGHUP the server for the changes to take effect, or use
# "pg_ctl reload".

# TYPE  DATABASE        USER            ADDRESS                 METHOD

# ====== Standard local connections (Unix-domain sockets) ======
# "local" is for Unix domain socket connections only
local   all             all                                     scram-sha-256

# ====== IPv4 local connections ======
host    all             all             127.0.0.1/32            scram-sha-256

# ====== IPv6 local connections ======
host    all             all             ::1/128                 scram-sha-256

# ====== Replication connections (optional, kept by default) ======
# Allow replication connections from localhost, by a user with the
# replication privilege.
local   replication     all                                     scram-sha-256
host    replication     all             127.0.0.1/32            scram-sha-256
host    replication     all             ::1/128                 scram-sha-256
